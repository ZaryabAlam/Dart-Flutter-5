// For-Loops
void main(){
  var i;
  for(i=1;i<=10;i++){
    print(i);
  } }


// For-Loop Table
void main(){
  var i;
  for(i=1; i<=10; i++){
    var j=4;
    var t=i*j;
    print("${j} x ${i} = ${t}");
    }   }


// While-Loop
void main(){
  var a=1;
  while(a<=10){
    print(a);
    a++;    
     }   }


// While-Loop Table
void main(){
  var i=1;
  while(i<=10){  
    var j=2;
    var t=j*i;
    print("${j} x ${i} = ${t}"); 
    i++;  
     }    }
